Linear Regression의 Hypothesis 와 cost 실습[2]

import tensorflow.compat.v1 as tf
import math

tf.disable_v2_behavior()

X and Y data(실제 데이터)

x_train = [1,2,3]
y_train = [1,2,3]

#텐서플로우의 Variable을 만든다. 즉, random한 값(난수)를 생성한다.
W = tf.Variable(tf.random.normal([1], name=’weight’))
b = tf.Variable(tf.random.normal([1], name=’bias’))

hypothesis = x_train * W + b

print(W)
print(b)
print(hypothesis)

#cost/loss function
#reduce_mean => 평균을 구하는 함수
#square => 제곱을 구하는 함수
cost = tf.reduce_mean(tf.square(hypothesis - y_train))

#GradientDescent
#위에서 구한 cost 값을 최소화 한다.
optimizer = tf.train.GradientDescentOptimizer(learning_rate=0.01)
train = optimizer.minimize(cost)

sess = tf.Session()
sess.run(tf.global_variables_initializer())

for step in range(2001):
sess.run(train)
if step % 20 == 0:
print(step, sess.run(cost), sess.run(W), sess.run(b))

#placeholder를 이용하여 Feed_dict로 값 넘겨주기
x_p = tf.placeholder(tf.float32)
y_p = tf.placeholder(tf.float32)

for step in range(2001):
cost_val, W_val, b_val, _ =
sess.run([cost, W, b, train,], feed_dict={x_p: [1,2,3], y_p: [1,2,3]})
if step % 20 == 0:
print(step, cost_val, W_val, b_val)
